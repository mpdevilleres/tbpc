basedir = os.path.abspath(os.path.dirname(__file__)) 

 

def id_reset_to(model=None, pk=2001): 

 

    table_name = model._meta.db_table 

 

    from django.db import connection 

    cursor = connection.cursor() 

    cursor.execute("ALTER SEQUENCE {0}_id_seq RESTART WITH {1};".format(table_name,pk)) 

 

def date_format(string): 

    try: 

        return dt.datetime.strptime(string, '%d/%m/%Y %H:%M') 

    except ValueError: 

        return dt.datetime.strptime('%s 00:00' % string, 

                                    '%d/%m/%Y %H:%M') 

 

def dump_contractors(file='contractors.csv'): 

    csvpath = os.path.join(basedir, file)

                record.name = row['name'] 

                record.remarks = row['remarks'] 

                record.profile = row['profile'] 

                record.short_hand = row['short_hand']

def dump_invoice(file='invoices_ho.csv'): 

    error_dict = { 

        'invoice': [], 

        'task': [], 

        'contractor': [] 

    } 

    csvpath = os.path.join(basedir, file) 

    with open(csvpath, encoding='mac_roman') as csvfile: 

        reader = csv.DictReader(csvfile) 

        for row in reader: 

            invoice = Invoice.objects.filter(invoice_no=row['invoice_no']).first() 

            task = Task.objects.filter(task_no=row['task_no_name']).first() 

            contractor = Contractor.objects.filter(id=int(row['contractor_id'])).first() 

 

            if invoice: 

                print('Invoice NO# %s Exist' % row['invoice_no']) 

                error_dict['invoice'].append(row['invoice_no']) 

                continue 

 

            elif task is None: 

                print('Task ID# %s not Exist' % row['task_no_name']) 

                error_dict['task'].append(row['task_no_name']) 

                continue 

 

            elif contractor is None: 

                print('Contractor ID# %s not Exist' % row['contractor_id']) 

                error_dict['contractor'].append(row['contractor_id']) 

                continue 

 

            else: 

                invoice = Invoice(task=task, 

                                  contractor=contractor) 

 

                invoice.region = row['region'] 

                invoice.invoice_no=row['invoice_no'] 

                invoice.contract_no=row['contract_no'] 

                invoice.revenue_amount=Decimal(row['revenue_amount']) 

                invoice.opex_amount=Decimal(row['opex_amount']) 

                invoice.capex_amount=Decimal(row['capex_amount']) 

                invoice.invoice_amount = Decimal(row['invoice_amount']) 

                invoice.invoice_date=date_format(row['invoice_date']) 

                invoice.invoice_cert_date=date_format(row['invoice_cert_date']) 

                invoice.received_date=date_format(row['received_date']) 

                invoice.signed_date=date_format(row['signed_date']) 

                invoice.start_date=date_format(row['start_date']) 

                invoice.end_date=date_format(row['end_date']) 

                invoice.rfs_date=date_format(row['rfs_date']) 

                invoice.sent_finance_date=date_format(row['sent_finance_date']) 

                invoice.cost_center='' 

                invoice.expense_code="-" 

                invoice.reference="-"
